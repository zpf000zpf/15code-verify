嵌入你自己的监控 / 采购流程：

from verify_core import Scanner, ScanConfig

scanner = Scanner(ScanConfig(
    base_url="https://api.some-reseller.com/v1",
    api_key="sk-xxx",
    claimed_model="claude-opus-4-7",
    tos_accepted=True,
))
report = scanner.run()
print(report.trust_score)
print(report.authenticity.likely_model)
